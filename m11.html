fetch('https://jsonplaceholder.typicode.com/posts', {

        body: 'Contenido del post'

3. Manejo Asíncrono

3.2 async/await

async function obtenerDatos() {
    try {
        const response = await fetch(url);
        const data = await response.json();
        return data;
    } catch (error) {
        console.error('Error:', error);
    }
}

5. Uso en React

5.1 useEffect

import { useState, useEffect } from 'react';

function App() {
    const [posts, setPosts] = useState([]);
    
    useEffect(() => {
        const fetchData = async () => {
            try {
                const response = await fetch(url);
                const data = await response.json();
                setPosts(data);
            } catch (error) {
                console.error(error);
            }
        };
        
        fetchData();
    }, []); // [] = Ejecutar solo al montar

    return (
        <div>
            {posts.map(post => (
                <div key={post.id}>{post.title}</div>
            ))}
        </div>
    );
}

6. Peticiones con Axios

6.1 Instalación

npm install axios

6.2 Uso Básico

import axios from 'axios';

// GET
axios.get('https://jsonplaceholder.typicode.com/posts/1')
    .then(response => console.log(response.data));

// POST
axios.post('https://jsonplaceholder.typicode.com/posts', {
    title: 'Nuevo post',
    body: 'Contenido'
});

6.4 Cancelación

const source = axios.CancelToken.source();

axios.get(url, {
    cancelToken: source.token
});

// Cancelar petición
source.cancel('Operación cancelada por el usuario');

8. Ejemplo Práctico

8.1 Obtener Países

function Paises() {
    const [paises, setPaises] = useState([]);
    
    useEffect(() => {
        axios.get('https://restcountries.com/v3.1/all')
            .then(response => setPaises(response.data))
            .catch(error => console.error(error));
    }, []);

    return (
        <div>
            {paises.map(pais => (
                <div key={pais.cca3}>
                    <h3>{pais.name.common}</h3>
                    <img src={pais.flags.png} alt="Bandera" />
                </div>
            ))}
        </div>
    );
}

9. Manejo de Estado Global

9.4 Context API

// Crear contexto
const ApiContext = createContext();

// Proveedor
function ApiProvider({ children }) {
    const [data, setData] = useState([]);
    
    const fetchData = async () => {
        const response = await axios.get(url);
        setData(response.data);
    };

    return (
        <ApiContext.Provider value={{ data, fetchData }}>
            {children}
        </ApiContext.Provider>
    );
}

// Uso en componentes
function Componente() {
    const { data } = useContext(ApiContext);
    return <div>{data.length} elementos</div>;
}
