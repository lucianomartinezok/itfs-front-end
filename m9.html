2. Componentes Contenedores vs Presentacionales

Contenedor (Smart)

class ListaUsuarios extends React.Component {
    state = { usuarios: [] };
    
    componentDidMount() {
        // Lógica para obtener datos
    }
    
    render() {
        return <ListaPresentacional items={this.state.usuarios} />;
    }
}

Presentacional (Dumb)

function ListaPresentacional({ items }) {
    return (
        <ul>
            {items.map(item => 
                <li key={item.id}>{item.nombre}</li>
            )}
        </ul>
    );
}

3. CSS en React

3.1 Estilos en Línea

const estilos = {
    boton: {
        padding: '10px 20px',
        backgroundColor: '#3498db',
        color: 'white'
    }
};

function Boton() {
    return <button style={estilos.boton}>Click</button>;
}

3.4 CSS Modules

// Boton.module.css
.boton {
    padding: 10px 20px;
    background: #3498db;
}

// Componente
import styles from './Boton.module.css';

function Boton() {
    return <button className={styles.boton}>Click</button>;
}

3.5 Styled-components

import styled from 'styled-components';

const Boton = styled.button`
    padding: 10px 20px;
    background: ${props => props.primario ? '#3498db' : '#95a5a6'};
    color: white;
`;

function App() {
    return <Boton primario>Click</Boton>;
}

6. Rutas en React

6.1 Instalación

npm install react-router-dom

6.3 Ejemplo Básico

import { BrowserRouter, Routes, Route, Link } from 'react-router-dom';

function App() {
    return (
        <BrowserRouter>
            <nav>
                <Link to="/">Inicio</Link>
                <Link to="/acerca">Acerca</Link>
            </nav>
            
            <Routes>
                <Route path="/" element={<Home />} />
                <Route path="/acerca" element={<Acerca />} />
            </Routes>
        </BrowserRouter>
    );
}

6.4 Rutas Dinámicas

<Route path="/usuario/:id" element={<Usuario />} />

// En componente Usuario:
import { useParams } from 'react-router-dom';

function Usuario() {
    const { id } = useParams();
    return <div>Usuario ID: {id}</div>;
}
