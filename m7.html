3. Manejo de Estado

useState Hook

import { useState } from 'react';

function Contador() {
    const [count, setCount] = useState(0);

    return (
        <div>
            <button onClick={() => setCount(c => c - 1)}>-</button>
            <span>{count}</span>
            <button onClick={() => setCount(c => c + 1)}>+</button>
        </div>
    );
}
